-- subjects and respective prof
SELECT 
	s.subject_code,
	s.subject_title,
	s.units,
	CONCAT(p.first_name,' ',p.last_name) AS "Professor Name",
	p.schedule
FROM
	subjects s
JOIN
	professors p
ON
	s.id = p.subject_id
WHERE
	s.id IS NOT NULL
GROUP BY
	s.id;

--subjects on a given course, year level and semester
SELECT 
	subject_code,
	subject_title 
FROM 
	subjects
WHERE
	course_id = 2 AND
	year_level = 3 AND
	term_id = 2;





--a section's subjects and profs (considering their year level and course) mali pa to

SELECT
	se.section_name,
	s.subject_code,
	s.subject_title,
	s.units,
	CONCAT(p.first_name,' ',p.last_name) AS "Professor Name",
	p.schedule
FROM
	subjects s
JOIN
	professors p
ON
	s.id = p.subject_id  -- This joins subjects (s) to professors (p)
JOIN
	sections se
ON
	se.year_level = s.year_level AND  -- This joins sections (se) to subjects (s)
	se.course_id = s.course_id AND
	se.term_id = s.term_id
WHERE
	s.id IS NOT NULL
GROUP BY
	s.id;
